func int fibo (int n);
var
    int resultado;
    int res1;
    int res2;
{
    if (n < 2) {
        resultado = n;
    } 
    else {
        res1 = fibo(n - 1);
        res2 = fibo(n - 2);
        resultado = res1 + res2;
    }
    return resultado;
}
