func int factorial (int n);
var 
    int resultado;
{
    if (n equal 1)
    {
        resultado = 1;
    }

    else 
    {
        resultado = n * factorial(n - 1);
    }

    return resultado;
}

func int fibo (int n);
var
    int resultado;
    int res1;
    int res2;
{
    if (n < 2) {
        resultado = n;
    } 
    else {
        res1 = fibo(n - 1);
        res2 = fibo(n - 2);
        resultado = res1 + res2;
    }
    return resultado;
}
